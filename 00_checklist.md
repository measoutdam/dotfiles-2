# Install python with ASDF
```
export ASDF_PYTHON_PATCH_URL="https://github.com/python/cpython/commit/8ea6353.patch?full_index=1"
asdf install python 3.9.15
asdf install python 2.7.10

# https://github.com/danhper/asdf-python#using-multiple-versions-of-python
asdf global python 3.9.15 2.7.13
# python -V
# Python 3.9.15
# 
# python3 -V
# Python 3.9.15
# 
# python2 -V
# Python 2.7.13
```
